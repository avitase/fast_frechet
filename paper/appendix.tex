\section{Scan and Fold}
\begin{algorithm}
    \caption{Scan operator for a binary function $f: \mathcal{T}_2 \times \mathcal{T}_1 \to \mathcal{T}_2$. The notation of using a backslash is borrowed from \citet{iverson62,iverson79}. Note that we prepend the initial value to the returned sequence.}
    \label{alg:scan}
    \begin{algorithmic}[1]
        \Function{$[f: \mathcal{T}_2 \times \mathcal{T}_1 \to \mathcal{T}_2]\backslash$}{$t_0: \mathcal{T}_2$, $x: \mathcal{T}_1^N$} $\to \mathcal{T}_2^{N+1}$
            \State $y: \mathcal{T}_2^{N+1}$
            \State $y_1 \gets t_0$
            \Statex
            \State $t: \mathcal{T}_2$ $\gets t_0$
            \For{$i \gets 1,N$}
                \State $t \gets f(t, x_i)$
                \State $y_{i+1} \gets t$
            \EndFor
            \Statex
            \State \Return $y$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}
    \caption{Fold operator for a binary function $f: \mathcal{T}_2 \times \mathcal{T}_1 \to \mathcal{T}_2$. The notation of using a slash is borrowed from \citet{iverson62,iverson79}.}
    \label{alg:fold}
    \begin{algorithmic}[1]
        \Function{$[f: \mathcal{T}_2 \times \mathcal{T}_1 \to \mathcal{T}_2]/$}{$t_0: \mathcal{T}_2$, $x: \mathcal{T}_1^N$} $\to \mathcal{T}_2$
            \State $t: \mathcal{T}_2$ $\gets t_0$
            \For{$i \gets 1,N$}
                \State $t \gets f(t, x_i)$
            \EndFor
            \Statex
            \State \Return $t$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
\section{Batched}
\begin{algorithm}
    \caption{Batched}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_maxmin}{$\alpha: \mathbb{R}^B$, $\chi: \mathbb{R}^{2 \times B}$} $\to \mathbb{R}^B$
            \State \Return $\max\{ \min\{ \alpha,\, \chi_1 \},\, \chi_2 \}$
        \EndFunction
        \Statex
        \Function{fr\'echet\_distance}{$[d_1: \mathbb{R}^{P_1 \times Q}$, $\ldots$, $d_N: \mathbb{R}^{P_N \times Q}]$, $B: \mathbb{N}_+$} $\to \mathbb{R}^N$
            \State $\mu: \mathbb{N}^B \gets [\min\{1, N\}, \ldots, \min\{B, N\}]$
            \State $\rho: \mathbb{N}^B \gets [1, \ldots, 1]$
            \Statex
            \State $\mathcal{V}: \mathbb{R}^{(Q+1) \times B} \gets [\max\!\backslash(0, d_{\min\{1, N\}, 1}) \;|\; \cdots \;|\; \max\!\backslash(0, d_{\min\{B, N\}, 1})]$ 
            \State $\mathcal{V}_1 \gets [\infty, \ldots, \infty]$
            \Statex
            \State $\delta_\mathrm{dF}: \mathbb{R}^N$
            \State $i: \mathbb{N} \gets 0$
            \While{$i < \max\{B, N\}$}
                \While{$\nexists j: \rho_j = P_{\mu_j}$}
                    \State $\rho \gets \rho + 1$
                    \Statex
                    \State $\mathcal{U}: \mathbb{R}^{Q \times B} \gets \min\{ \mathcal{V}_{1 \ldots Q},\, \mathcal{V}_{2 \ldots Q+1} \}$
                    \State $\mathcal{D}: \mathbb{R}^{Q \times B} \gets [d_{\mu_1,\, \min\{\rho_1, P_{\mu_1}\}} \;|\; \cdots \;|\; d_{\mu_B,\, \min\{\rho_B, P_{\mu_B}\}}]$
                    \State $\mathcal{V} \gets$ \Call{fr\'echet\_maxmin$\backslash$}{$\infty,\, [[\mathcal{U}_1, \mathcal{D}_1], \ldots, [\mathcal{U}_Q, \mathcal{D}_Q]]$}
                \EndWhile
                \Statex
                \For{$j \gets \{j \,|\, \rho_j = P_{\mu_j} \}$}
                    \State ${\delta_\mathrm{dF}}_{\mu_j} \gets \mathcal{V}_{Q+1,j}$
                    \State $i \gets i + 1$
                    \Statex
                    \State $\rho_j \gets \rho_j + 1$
                    \If{$i + B \le N$}
                        \State $\mu_j = i + B$
                        \State $\rho_j = 0$
                        \Statex
                        \State $\mathcal{V}_{:,j} \gets \max\!\backslash(0, d_{\mu_j,1})$ 
                        \State $\mathcal{V}_{1,j} \gets \infty$
                    \EndIf
                \EndFor
            \EndWhile
            \Statex
            \State \Return $\delta_\mathrm{dF}$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}
    \caption{DTW}
    \begin{algorithmic}[1]
        \Function{dtw\_min}{$a: \mathbb{R}$, $x: \mathbb{R}^2$} $\to \mathbb{R}$
            \State \Return $\min\{ a,\, x_1 \} + x_2$
        \EndFunction
        \Statex
        \Function{dtw\_next}{$a: \mathbb{R}^{Q+1}$, $x: \mathbb{R}^{Q}$} $\to \mathbb{R}^{Q+1}$
            \State $u: \mathbb{R}^Q \gets \min\{ a_{1 \ldots Q},\, a_{2 \ldots Q+1} \}$
            \State \Return \Call{dtw\_min$\backslash$}{$\infty,\, [u \;|\; x]$}
        \EndFunction
        \Statex
        \Function{dtw\_distance}{$d: \mathbb{R}^{P \times Q}$} $\to \mathbb{R}$
            \State $v: \mathbb{R}^{Q+1} \gets +\!\backslash(0, d_1)$
            \State $v_1 \gets \infty$
            \Statex
            \State \Return $[$\Call{dtw\_next$/$}{$v, d_{2 \ldots P}$}$]_{Q+1}$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
