\section{Introduction}
The Fr\'echet distance was introduced by Maurice Fr\'echet in 1906 \citep{frechet06} and is well known as a fundamental metric in abstract spaces in the field of mathematics and computational geometry.
Over the years, it has found applications in various domains, e.g., as a distance measure between probability distributions it is equivalent to the Wasserstein distance \citep{kantorovich60,wasserstein69} for the Euclidean distance.
\citet{dowson82} demonstrated that for normal distributions the Fr\'echet distance can be calculated explicitly and since then it is used a basis of various evaluation metrics such as the Fr\'echet inception distance \citep{heusel17}, Fr\'echet audio distance \citep{kilgour19}, or the Fr\'echet ChemNet distance \citep{preuer18}.
In the early 1990s Alt and Godau were the first to apply the Fr\'echet distance in measuring the similarity of polygonal curves \citep{alt92,alt95}.
However, without strong assumption about the polygonal curves, the distance cannot be calculated directly here.
Finding efficient algorithms for estimating the Fr\'echet distance in this setting therefore is still an activate topic in the research community.

Formally, let $P$ and $Q$ be to curves in a metric space $\mathcal{S}$.
Then, the Fr\'echet distance $\delta_F$ between $P$ and $Q$ is defined as the infimum over all possible continuous, non-decreasing, surjections $\alpha: [0, 1] \to [0, 1]$ and $\beta: [0, 1] \to [0, 1]$ of the maximum over all $t \in [0, 1]$ of the distance in $\mathcal{S}$ between $P(\alpha(t))$ and $Q(\beta(t))$,
\begin{equation*}
    \delta_F(P, Q) = \inf_{\alpha,\beta} \; \max_{t \in [0, 1]} \left\| P(\alpha(t)) - Q(\beta(t)) \right\|_\mathcal{S},
\end{equation*}
where $\| \bullet \|_\mathcal{S}$ is the distance functions of $\mathcal{S}$.
Informally, this measure is often described as the minimal leash length, measured by $\| \bullet \|_\mathcal{S}$, of a man walking his dog if both are moving on trajectories $P(\alpha(t))$ and $Q(\beta(t))$ respectively, where $t$ is interpreted as a measure of time.
In this interpretation, the restriction that $\alpha(t)$ and $\beta(t)$ be non-decreasing means that neither the dog nor its owner can backtrack.
Relaxing this constraint, defines the so-called weak Fr\'echet distance \citep{alt95} that is always greater or equal than the Hausdorff distance, yet smaller or equal than the (continuous) Fr\'echet distance.

Although non-straightforward to compute, the Fr\'echet distance has been used successfully in various fields such as curve simplification \citep{agarwal05,bereg08}, clustering \citep{buchin08b}, and map-matching \citep{alt03a,driemel12}.
The Fr\'echet distance also has applications in matching biological sequences \citep{wylie12}, analysing tracking data \citep{brakatsoulas05,buchin08a}, and matching coastline data \citep{mascret06}.

\citet{bringmann14} showed that Fr\'echet distance has no strongly subquadratic algorithm unless SETH fails and even bringing down the complexity of robust algorithms close to a quadratic algorithm has proven to be challenging in practice \citep{bringmann19} and are often complex to implement.
In particular, almost all of the analyses and applications of the continuous Fr\'echet distance either explicitly or implicitly rely on the Euclidean distance for $\| \bullet \|_\mathcal{S}$ which makes the corresponding pathing in \textit{free-space} a convex optimization problem.
Extending the results to more general metrics is not straightforward as one quickly looses auxiliary properties such as the convexity in \textit{free-space}.

One way of overcoming these challenges is to estimate the continuous Fr\'echet distance with a discrete approximation.
\citet{eiter94} defined the discrete Fr\'echet distance $\delta_{dF}$ between two polygonal chains, proposed the simple algorithm Alg.~\ref{alg:vanilla} of (exactly) quadratic complexity, and showed that the difference between the continuous Fr\'echet distance and its discrete variant is bound by the sample width of the trajectories,
\begin{equation*}
    \delta_F(P, Q) \;\le\; \delta_{dF}(P, Q) \;\le\; \delta_F(P, Q) + \max\{ \varepsilon_P, \varepsilon_Q \},
\end{equation*}
where $\varepsilon_P$ and $\varepsilon_Q$ are the largest distance between adjacent points on $P$ and $Q$, respectively.

\begin{algorithm}
    \caption{Vanilla}
    \label{alg:vanilla}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_distance}{$p$: $\mathbb{R}^{P \times D}$, $q$: $\mathbb{R}^{Q \times D}$, $f$: $\mathbb{R}^D \times \mathbb{R}^D \mapsto \mathbb{R}$} $\to \mathbb{R}$
            \State $M: \mathbb{R}^{P \times Q}$ $\gets -1$
            \Statex
            \Function{eval}{$i: \mathbb{N}$, $j: \mathbb{N}$} $\to \mathbb{R}$
                \If{$M_{ij} > -1$}
                    \State \Return $M_{ij}$
                \EndIf
                \Statex
                \State $d: \mathbb{R}$ $\gets$ \Call{$f$}{$p_i, q_j$}
                \If{$i = 1 \;\land\; j = 1$}
                    \State $M_{ij} \gets d$
                \ElsIf{$i > 1 \;\land\; j = 1$}
                    \State $M_{ij} \gets \max\{\Call{eval}{i - 1, 1},\, d\}$
                \ElsIf{$i = 1 \;\land\; j > 1$}
                    \State $M_{ij} \gets \max\{\Call{eval}{1, j - 1},\, d\}$
                \Else
                    \State $M_{ij} \gets \max\{\min\{\Call{eval}{i - 1, j},\, \Call{eval}{i - 1, j - 1},\, \Call{eval}{i, j - 1}\},\, d\}$
                \EndIf
                \Statex
                \State \Return $M_{ij}$
            \EndFunction
            \Statex
            \State \Return \Call{eval}{$P - 1, Q - 1$}
        \EndFunction
    \end{algorithmic}
\end{algorithm}

    \caption{No recursion}
    \label{alg:no_recursion}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_distance}{$p: \mathbb{R}^{P \times D}$, $q: \mathbb{R}^{Q \times D}$, $f: \mathbb{R}^D \times \mathbb{R}^D \mapsto \mathbb{R}$} $\to \mathbb{R}$

            \For{$i \gets 1,P$}
                \For{$j \gets 1,Q$}

                    \ElsIf{$i > 1 \;\land\; j = 1$}
                        \State $M_{ij} \gets \max\{M_{i-1,1},\, d\}$

                        \State $M_{ij} \gets \max\{M_{1,j-1},\, d\}$
                    \Else
                        \State $M_{ij} \gets \max\{\min\{M_{i-1,j},\, M_{i-1,j-1},\, M_{i,j-1}\},\, d\}$
                    \EndIf
                \EndFor
            \EndFor
