taken from \citep{bereg08}:
Fr\'echet distance was first defined by Maurice Fr\'echet in 1906 \citep{frechet06}.
While known as a famous distance measure in the field of mathematics (more specifically, abstract spaces), it was Alt and Godau who first applied it in measuring the similarity of polygonal curves in early 1990s \citep{alt92,alt95}.
In 1994, Eiter and Mannila defined the discrete Fr\'echet distance between two polygonal chains A and B (in any fixed dimensions) \citep{eiter94}.
Recently, Jiang, Xu and Zhu applied the discrete Fr\'echet distance in aligning the backbones of proteins (which is called the protein structure-structure alignment problem) \citep{jiang08}.
In fact, in this application the discrete Fr\'echet distance makes more sense as the backbone of a protein is simply a polygonal chain in 3D, with each vertex being the alpha-carbon atom of a residue.
So if the (continuous) Fr\'echet distance is realized by an alpha-carbon atom and some other point which does not represent an atom, it is not meaningful biologically.
Jiang, et al.\ showed that given two planar polygonal chains the minimum discrete Fr\'echet distance between them, under both translation and rotation, can be computed in polynomial time.
They also applied some ideas therein to design an efficient heuristic for the original protein structure-structure alignment problem in 3D.

\begin{algorithm}
    \caption{Vanilla}
    \label{alg:frechet_vanilla}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_distance}{$p$: $\mathbb{R}^{P \times D}$, $q$: $\mathbb{R}^{Q \times D}$, $f$: $\mathbb{R}^D \times \mathbb{R}^D \mapsto \mathbb{R}$} $\to \mathbb{R}$
            \State $M: \mathbb{R}^{P \times Q}$ $\gets -1$
            \Statex
            \Function{eval}{$i: \mathbb{N}$, $j: \mathbb{N}$} $\to \mathbb{R}$ \Comment{Capture $p$, $q$, $f$, and $M$}
                \If{$M_{ij} > -1$}
                    \State \Return $M_{ij}$
                \EndIf
                \Statex
                \State $d: \mathbb{R}$ $\gets$ \Call{$f$}{$p_i, q_j$}
                \If{$i = 1 \;\land\; j = 1$}
                    \State $M_{ij} \gets d$
                \ElsIf{$i > 1 \;\land\; j = 1$}
                    \State $M_{ij} \gets \max\{\Call{eval}{i - 1, 1},\, d\}$
                \ElsIf{$i = 1 \;\land\; j > 1$}
                    \State $M_{ij} \gets \max\{\Call{eval}{1, j - 1},\, d\}$
                \Else
                    \State $M_{ij} \gets \max\{\min\{\Call{eval}{i - 1, j},\, \Call{eval}{i - 1, j - 1},\, \Call{eval}{i, j - 1}\},\, d\}$
                \EndIf
                \Statex
                \State \Return $M_{ij}$
            \EndFunction
            \Statex
            \State \Return \Call{eval}{$P - 1, Q - 1$}
        \EndFunction
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}
    \caption{No recursion}
    \label{alg:frechet_no_recursion}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_distance}{$p: \mathbb{R}^{P \times D}$, $q: \mathbb{R}^{Q \times D}$, $f: \mathbb{R}^D \times \mathbb{R}^D \mapsto \mathbb{R}$} $\to \mathbb{R}$
            \State $M: \mathbb{R}^{P \times Q}$ $\gets -1$
            \Statex
            \For{$i \gets 1,P$}
                \For{$j \gets 1,Q$}
                    \State $d: \mathbb{R}$ $\gets$ \Call{$f$}{$p_i, q_j$}
                    \If{$i = 1 \;\land\; j = 1$}
                        \State $M_{ij} \gets d$
                    \ElsIf{$i > 1 \;\land\; j = 1$}
                        \State $M_{ij} \gets \max\{M_{i-1,1},\, d\}$
                    \ElsIf{$i = 1 \;\land\; j > 1$}
                        \State $M_{ij} \gets \max\{M_{1,j-1},\, d\}$
                    \Else
                        \State $M_{ij} \gets \max\{\min\{M_{i-1,j},\, M_{i-1,j-1},\, M_{i,j-1}\},\, d\}$
                    \EndIf
                \EndFor
            \EndFor
            \Statex
            \State \Return $M_{P-1,Q-1}$   
        \EndFunction
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}
    \caption{Linear memory}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_distance}{$p: \mathbb{R}^{P \times D}$, $q: \mathbb{R}^{Q \times D}$, $f: \mathbb{R}^D \times \mathbb{R}^D \mapsto \mathbb{R}$} $\to \mathbb{R}$
            \State $d_1: \mathbb{R}^Q \gets$ \Call{$f$}{$p_1, q$}
            \Statex
            \State $v: \mathbb{R}^{Q+1} \gets \max\backslash(0, d_1)$
            \State $v_1 \gets \infty$
            \Statex
            \For{$i \gets 2,P$}
                \State $d_i: \mathbb{R}^Q \gets$ \Call{$f$}{$p_i, q$}
                \State $u: \mathbb{R}^Q \gets \min\{ v_{1 \ldots Q},\, v_{2 \ldots Q+1} \}$
                \Statex
                \For{$j \gets 1,Q$}
                    \State $v_{j+1} \gets \max\{ \min\{ v_j,\, u_j \},\, d_{ij} \}$
                \EndFor
            \EndFor
            \Statex
            \State \Return $v_{Q+1}$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

        \EndFunction
        \Statex

\begin{algorithm}
    \caption{Levenshtein}
    \begin{algorithmic}[1]
        \Function{levenshtein\_min}{$a: \mathbb{N}$, $x: \mathbb{N}^2$} $\to \mathbb{N}$
            \State \Return $\min\{ \min\{a,\, x_1 \} + 1, x_2 \}$
        \EndFunction
        \Statex
        \Function{levenshtein\_distance}{$d: \mathbb{N}^{P \times Q}$} $\to \mathbb{N}$
            \State $v: \mathbb{N}^{Q+1} \gets \min(\mathbf{1}_Q, d_1) \,+\,[0, 1, \ldots, Q-1]$
            \Statex
            \For{$i \gets 2,P$}
                \State $u: \mathbb{N}^Q \gets \min\{ v_{1 \ldots Q},\, v_{2 \ldots Q+1} \}$
                \State $v \gets$ \Call{levenshtein\_min$\backslash$}{$i+1,\, [u \;|\; d_i + v_{1 \ldots Q}]$} 
            \EndFor
            \Statex
            \State \Return $v_{Q+1}$
        \EndFunction
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}
    \caption{Batched}
    \begin{algorithmic}[1]
        \Function{fr\'echet\_maxmin}{$\alpha: \mathbb{R}^B$, $\chi: \mathbb{R}^{2 \times B}$} $\to \mathbb{R}^B$
            \State \Return $\max\{ \min\{ \alpha,\, \chi_1 \},\, \chi_2 \}$
        \EndFunction
        \Statex
        \Function{fr\'echet\_distance}{$[d_1: \mathbb{R}^{P_1 \times Q}$, $\ldots$, $d_N: \mathbb{R}^{P_N \times Q}]$, $B: \mathbb{N}_+$} $\to \mathbb{R}^N$
            \State $\mu: \mathbb{N}^B \gets [1, \ldots, B]$
            \State $\rho: \mathbb{N}^B \gets [1, \ldots, 1]$
            \State $w: \mathbb{R}^N$
            \Statex
            \State $\mathcal{V}: \mathbb{R}^{(Q+1) \times B} \gets [\max\backslash(0, d_{1,1}) \;|\; \cdots \;|\; \max\backslash(0, d_{B,1})]$ 
            \State $\mathcal{V}_1 \gets \infty$
            \Statex
            \State $i: \mathbb{N} \gets 0$
            \While{$i < N$}
                \While{$\min\{ P_{\mu_1} - \rho_1, \ldots, P_{\mu_B} - \rho_B \} > 0$}
                    \State $\rho \gets \rho + 1$
                    \Statex
                    \State $\mathcal{U}: \mathbb{R}^{Q \times B} \gets \min\{ \mathcal{V}_{1 \ldots Q},\, \mathcal{V}_{2 \ldots Q+1} \}$
                    \State $\mathcal{D}: \mathbb{R}^{Q \times B} \gets [d_{\mu_1,\, \rho_1} \;|\; \cdots \;|\; d_{\mu_B,\, \rho_B}]$
                    \State $\mathcal{V} \gets$ \Call{fr\'echet\_maxmin$\backslash$}{$\infty,\, [[\mathcal{U}_1, \mathcal{D}_1], \ldots, [\mathcal{U}_Q, \mathcal{D}_Q]]$}
                \EndWhile
                \Statex
                \For{$j \gets \{j \,|\, \rho_j = P_{\mu_j} \}$}
                    \State $i \gets i + 1$
                    \Statex
                    \State $\mu_j = \max\{i + 1, N\}$
                    \State $\rho_j = P_{\mu_j}$
                    \State $w_{\mu_j} \gets \mathcal{V}_{Q+1,j}$
                    \Statex
                    \State $\mathcal{V}_{:,j} \gets \max\backslash(0, d_{\mu_j,1})$ 
                    \State $\mathcal{V}_{:,1} \gets \infty$
                \EndFor
            \EndWhile
            \Statex
            \State \Return $w$
        \EndFunction
    \end{algorithmic}
\end{algorithm}
